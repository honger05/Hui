````js
var slide = new Slide({
    element: '#slide-demo-3',
    effect: 'scrollx',
    hasTriggers: false
}).render();

// 自定义 prev/next
$("#slide-demo-3 #prev").click(function(e) {
    e.preventDefault();
    slide.prev();
});

$("#slide-demo-3 #next").click(function(e) {
    e.preventDefault();
    slide.next();
});
````
